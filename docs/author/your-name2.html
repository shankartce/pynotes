    while count < limit:
        yield a

def factorial(n):
    if n < 0:
        raise ValueError("n must be a non-negative integer")
    if n in (0, 1):
        return 1
    return n * factorial(n - 1)

for value in range(6):
    print(f"factorial({value}) = {factorial(value)}")
